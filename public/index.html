Example Usage

# Get all students
curl https://student-database-management-node.vercel.app/api/students-final

# Add a student
curl -X POST https://student-database-management-node.vercel.app/api/students-final \
  -H "Content-Type: application/json" \
  -d '{"Name":"John Doe","Mark":85,"Active":"T"}'

# Update a student
curl -X PUT https://student-database-management-node.vercel.app/api/students-final?id=1 \
  -H "Content-Type: application/json" \
  -d '{"Mark":95}'
